1function [joint_angle, osc_x] = CPGNetWork( acc_roll,slope_alpha, v, T, osc_x, tfinal, phi, theta0, beta, phiRB, phiLB, alpha, a, k1 ,k2)
 2%% 参数
 3%体态参数
 4length = 0.9;                   %体长  m
 5width = 0.5;                    %体宽  m
 6l = 0.3;                        %腿节长度  m
 7%步态参数
 8omiga_sw=1/(beta*T);            %摆动相频率
 9if(acc_roll>0.959595959)
10    h = acc_roll*sin(atan(width/length))/(2*omiga_sw^2*sqrt(1-acc_roll^2/(4*omiga_sw^4)));                       %摆动腿足端高度  m
11else
12    h=0.02;
13end
14Ak = acos((l*cos(theta0)-h)/l)-theta0;       %髋关节幅值
15Ah = asin((beta*v*T)/(4*l/cos(theta0)));     %膝关节幅值
16%振荡器参数
17miu=Ah^2;                       %振荡器幅值 A=sqrt(miu)
18%反馈项参数
19%前庭反射
20delte_theta=phi*k2*k1*slope_alpha;  %原平衡位置改变量
21U = [-(delte_theta/2-k1*slope_alpha); delte_theta;     %外部反馈项(feed) 对应x1 y1 x2 y2 x3 y3 x4 y4
22     -(delte_theta/2-k1*slope_alpha); delte_theta; 
23     -(delte_theta/2-k1*slope_alpha); delte_theta; 
24     -(delte_theta/2-k1*slope_alpha); delte_theta;
25     0;0;0;0;0;0;0;0];
26%% 初值
27t0 = 0;              % 响应时间域（Time region of response）
28%% 求解
29% 求解Hopf振荡器微分方程的数值解（Numerical solution of differential equation）
30[t, x] = ode45('CPGNetWork_hopf_oscillator_with_lateral_impact', [t0, tfinal], osc_x,[],alpha, beta, miu, omiga_sw, a, phiRB, phiLB, U);
31%振荡器输出
32for i = 1:16
33    osc_x(i)=x(size(x,1), i);
34end
35%修改振荡器输出y为膝关节，振荡器输出x为髋关节
36for i=2:2:9
37    for j=1:size(x,1)
38        if x(j,i)<=U(i)
39            x(j,i)=-phi*(Ak/Ah)*(x(j,i)-U(i))+U(i);
40        else
41            x(j,i)=U(i);
42        end
43    end
44end
45x_out=x;
46% 振荡器关节输出
47joint_angle=[x_out(size(x_out,1), 1),x_out(size(x_out,1), 2),x_out(size(x_out,1), 3),x_out(size(x_out,1), 4),x_out(size(x_out,1), 5),x_out(size(x_out,1), 6),x_out(size(x_out,1), 7),x_out(size(x_out,1), 8),x_out(size(x_out,1), 9),x_out(size(x_out,1), 10),x_out(size(x_out,1), 11),x_out(size(x_out,1), 12),x_out(size(x_out,1), 13),x_out(size(x_out,1), 14),x_out(size(x_out,1), 15),x_out(size(x_out,1), 16)];
48fprintf('LF  hip-pitch:%f  knee:%f  hip-roll:%f  ', joint_angle(1),joint_angle(2),joint_angle(9));
49fprintf('RF  hip-pitch:%f  knee:%f  hip-roll:%f  ', joint_angle(3),joint_angle(4),joint_angle(11));
50fprintf('RB  hip-pitch:%f  knee:%f  hip-roll:%f  ', joint_angle(5),joint_angle(6),joint_angle(13));
51fprintf('RL  hip-pitch:%f  knee:%f  hip-roll:%f\n', joint_angle(7),joint_angle(8),joint_angle(15));
52end
